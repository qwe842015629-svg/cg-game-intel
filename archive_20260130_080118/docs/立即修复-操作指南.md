```javascript
// 验证热门游戏数量

  .then(r => r.json())
  .then(data => {
    const hotGames = data.find(d => d.sectionKey === 'hot_games')
    const latestNews = data.find(d => d.sectionKey === 'latest_news')
    console.log('热门游戏配置:', hotGames?.config.display_count)
    console.log('最新资讯配置:', latestNews?.config.display_count)
  })
```
