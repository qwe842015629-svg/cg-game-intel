## 🎯 最佳实现方案：动态表单选择

### 核心思路

在`HomeLayoutAdmin`中根据`section_key`动态选择对应的表单，每个板块看到自己专属的配置字段。

### 实现步骤

#### 步骤1：在admin.py中修改HomeLayoutAdmin

```python
from main.forms import (
    BannerSectionForm,
    HotGamesSectionForm,
    LatestNewsSectionForm,
    FeaturesSectionForm,
    CategoriesSectionForm,
    BaseConfigForm
)

class HomeLayoutAdmin(admin.ModelAdmin):
    # 移除固定的form配置
    # form = HomeLayoutAdminForm  # 删除这行
    
    def get_form(self, request, obj=None, **kwargs):
        \"\"\"根据板块类型动态选择表单\"\"\"
        if obj:
            # 表单映射
            FORM_MAPPING = {
                'banner_section': BannerSectionForm,      # 轮播图
                'hot_games': HotGamesSectionForm,         # 热门游戏
                'latest_news': LatestNewsSectionForm,     # 最新资讯
                'features': FeaturesSectionForm,          # 核心特性
                'categories': CategoriesSectionForm,      # 游戏分类
            }
            kwargs['form'] = FORM_MAPPING.get(obj.section_key, BaseConfigForm)
        else:
            # 新建时使用基础表单
            kwargs['form'] = BaseConfigForm
        
        return super().get_form(request, obj, **kwargs)
    
    def get_fieldsets(self, request, obj=None):
        \"\"\"根据板块类型动态调整字段分组\"\"\"
        base_fieldsets = [
            ('板块基本信息', {
                'fields': ('section_key', 'section_name'),
            }),
            ('显示设置', {
                'fields': ('is_enabled', 'sort_order'),
                'description': '<b>排序说明：</b>数字越小越靠前'
            }),
        ]
        
        if not obj:
            return base_fieldsets + [
                ('统计信息', {
                    'fields': ('view_count', 'created_at', 'updated_at'),
                    'classes': ('collapse',),
                }),
            ]
        
        # 根据板块类型添加专属配置区域
        if obj.section_key == 'banner_section':
            config_fieldset = ('🎪 轮播图配置', {
                'fields': (
                    'config_title',
                    'config_subtitle',
                    'config_auto_play',
                    'config_interval',
                    'config_show_indicators',
                    'config_show_arrows',
                    'config_pause_on_hover',
                ),
                'classes': ('wide',),
            })
        elif obj.section_key == 'hot_games':
            config_fieldset = ('🔥 热门游戏配置', {
                'fields': (
                    'config_title',
                    'config_subtitle',
                    'config_icon',
                    'config_display_count',
                    'config_show_more_button',
                    'config_layout',
                ),
                'classes': ('wide',),
            })
        elif obj.section_key == 'latest_news':
            config_fieldset = ('📰 最新资讯配置', {
                'fields': (
                    'config_title',
                    'config_subtitle',
                    'config_icon',
                    'config_display_count',
                    'config_show_category',
                    'config_show_author',
                    'config_show_date',
                    'config_show_more_button',
                ),
                'classes': ('wide',),
            })
        elif obj.section_key == 'features':
            config_fieldset = ('✨ 核心特性配置', {
                'fields': (
                    'config_title',
                    'config_subtitle',
                    ('config_feature_1_icon', 'config_feature_1_title', 'config_feature_1_desc'),
                    ('config_feature_2_icon', 'config_feature_2_title', 'config_feature_2_desc'),
                    ('config_feature_3_icon', 'config_feature_3_title', 'config_feature_3_desc'),
                ),
                'classes': ('wide',),
            })
        elif obj.section_key == 'categories':
            config_fieldset = ('🎮 游戏分类配置', {
                'fields': (
                    'config_title',
                    'config_subtitle',
                    'config_show_all_category',
                    'config_layout_style',
                ),
                'classes': ('wide',),
            })
        else:
            config_fieldset = ('板块配置', {
                'fields': ('config_title', 'config_subtitle'),
                'classes': ('wide',),
            })
        
        return base_fieldsets + [
            config_fieldset,
            ('📊 配置预览', {
                'fields': ('config_preview',),
                'classes': ('collapse',),
            }),
            ('统计信息', {
                'fields': ('view_count', 'created_at', 'updated_at'),
                'classes': ('collapse',),
            }),
        ]
```
