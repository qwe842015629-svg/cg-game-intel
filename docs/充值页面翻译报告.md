```typescript
  // ...
]
```

**建议优化方案**：

#### 方案1：使用翻译键
```typescript
const rechargeAmounts: RechargeAmount[] = [

]

// 模板中：

```

```typescript
{{ bonus.amount }}{{ $t(bonus.currencyType) }}
```
